title_favored_son = {
	grant_limit = 1
	opinion_effect = 10
	monthly_salary = 0
	monthly_prestige = 0
	allow = {
		is_female = no
		is_adult = no
		martial = 3
		diplomacy = 3
		stewardship = 3
		intrigue = 3
		learning = 3
		NOT = {
			trait = bastard
			trait = incapable
			trait = inbred
		}
		is_child_of = FROM
		dynasty = FROM
		is_liege_or_above = FROM
	}
	gain_effect = {
		add_trait = favored_son
	}
	lose_effect = {
		remove_trait = favored_son
	}
	message = no
}

title_ladys_champion = {
	dignity = 0.2
	realm_in_name = yes
	grant_limit = 1
	opinion_effect = 20
	
	monthly_salary = 0.1
	monthly_prestige = 0.075
	
	allow = {
		is_adult = yes
		is_female = no
		prisoner = no
		NOT = {
			trait = incapable
			health_traits = 1
			has_character_flag = retired_champion
		}
		OR = {
			trait = misguided_warrior
			trait = tough_soldier
			trait = skilled_tactician
			trait = brilliant_strategist
		}
		OR = {
			FROM = { ai = no }
			AND = {
				opinion = { who = FROM value = 0 }
				reverse_opinion = { who = FROM value = 0 }
			}
		}
		FROM = {
			is_feudal = yes
			is_adult = yes
			is_female = yes
			NOT = { # already have a Champion title
				religion = baltic_pagan_reformed
				religion = baltic_pagan
				religion = finnish_pagan_reformed
				religion = finnish_pagan
				religion = slavic_pagan_reformed
				religion = slavic_pagan
				religion = west_african_pagan_reformed
				religion = west_african_pagan
				religion = aztec_pagan_reformed
				religion = aztec_pagan
			}
		}
	}
	
	gain_effect = {
	}
	lose_effect = {
	}

	message = yes
}
