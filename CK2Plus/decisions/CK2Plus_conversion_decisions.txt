	convert_to_capital_religion = {
		potential = {
			is_playable = yes

				NOT = { religion = ROOT }

						liege = {

						}
					}
				}
			}
			NOT = { has_character_flag = converted_religion }
			NOT = {
				AND = {
					has_dlc = "Charlemagne"

			}
		}
		allow = {
			NOT = { trait = zealous }
		}
		effect = {
			piety = -100
			if = {
				limit = { higher_tier_than = BARON }
				religion_authority = {
					modifier = ruler_converted_from
				}
			}
			capital_scope = {
				reverse_religion = ROOT
			}
			if = {
				limit = { higher_tier_than = BARON }
				hidden_tooltip = {
					religion_authority = {
						modifier = ruler_converted_to
					}
				}
			}
			set_character_flag = converted_religion
		}
		revoke_allowed = {
			always = no
		}
		ai_will_do = {
			factor = 0
		}
	}
